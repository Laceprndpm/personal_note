如果存在某一袋购买方案是用到了大量的 `1` 和 `2` ，肯定有将 `1` 和 `2` 换成 `3` 的更优的方案，所以枚举 `1` 和 `2` 的数量，然后暴力模拟。
然后注意实现细节中， $c \ge 0$ 的限制。
```cpp showLineNumbers
void solve()
{
    int n;
    cin >> n;
    vector<i64> arr(n);
    for (int i = 0; i < n; i++) {
        cin >> arr[i];
    }
    auto check = [&](int num1, int num2) -> pair<bool, i64> {
        i64 mxval = 0;
        for (int i = 0; i < n; i++) {
            i64 minval = INF;
            for (int n1 = 0; n1 <= num1; n1++) {
                for (int n2 = 0; n2 <= num2; n2++) {
                    i64 c = arr[i] - n1 - n2 * 2;
                    if (c % 3 == 0 && c >= 0) {
                        minval = min(minval, c / 3);
                    }
                }
            }
            if (minval == INF) {
                return {0, 0};
            } else {
                mxval = max(mxval, minval);
            }
        }
        return {1, mxval};
    };
    i64 ans = INF;
    for (int i = 0; i <= 6; i++) {
        for (int j = 0; j <= 6; j++) {
            auto [ok, val] = check(i, j);
            if (!ok) continue;
            ans = min(ans, val + i + j);
        }
    }
    cout << ans << '\n';
}
```
