$$
\boxed{
\begin{align}
\text{形式一：}f(n) = \sum_{d|n} g(d),\ \text{那么有 }g(n) = \sum_{d|n} \mu(\frac{d}{n})f(d) \\
\text{形式二：}f(n) = \sum_{n | d}g(d),\ \text{那么有 }g(n) = \sum_{n | d}\mu(\frac{d}{n})f(d)
\end{align}
}
$$
